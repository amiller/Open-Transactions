$ eval `brew --config | grep HOMEBREW_PREFIX | sed 's/: /=/'`
$ sudo bash -c 'echo '$HOMEBREW_PREFIX/share/aclocal' >> `aclocal --print-ac-dir`/dirlist'
(optional, for non-global brew installs)
$ export PATH=$PATH:$HOMEBREW_PREFIX/bin

# install dependencies
# homebrew ones
$ brew install cmake protobuf msgpack zeromq boost

# homebrew doesn't pack openssl 1.0 yet
$ mkdir -p ~/Scratch/Sources; cd ~/Scratch/Sources
$ wget http://openssl.org/source/openssl-1.0.0g.tar.gz
$ tar xf openssl-1.0.0g.tar.gz
$ cd openssl-1.0.0g
$ ./Configure darwin64-x86_64-cc --prefix=$HOME/.local
$ make 
$ make install

# now chaiscript
$ cd ..
$ wget https://github.com/downloads/ChaiScript/ChaiScript/chaiscript-3.1.0-Source.tar.bz2
$ tar xf chaiscript-3.1.0-Source.tar.bz2 
$ cd chaiscript-3.1.0-Source
$ cmake -DBOOST_ROOT=$HOMEBREW_PREFIX -DCMAKE_INSTALL_PREFIX=$HOMEBREW_PREFIX
$ make
$ make install
